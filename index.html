# % de valores faltantes  em cada base
def missing_table(df, name):
    mis = df.isna().sum().sort_values(ascending=False)
    mis = mis[mis > 0]
    if mis.empty:
        print(f"\n{name}: sem valores faltantes ✅")
        return
    pct = (mis / len(df) * 100).round(2)
    out = pd.DataFrame({"faltantes": mis, "%": pct})
    print(f"\n{name}: colunas com faltantes (top 15)")
    display(out.head(15))

## 7 Faixas simples de renda × taxa de inadimplência

query = """
SELECT
  CASE
    WHEN a.AMT_INCOME_TOTAL/80 < 1250 THEN '<1.25k USD'
    WHEN a.AMT_INCOME_TOTAL/80 < 2500 THEN '1.25k-2.5k USD'
    WHEN a.AMT_INCOME_TOTAL/80 < 5000 THEN '2.5k-5k USD'
    ELSE '5k+ USD'
  END AS faixa_renda_usd,
  COUNT(*)             AS clientes,
  AVG(a.TARGET*1.0)    AS taxa_inad
FROM application_train a
WHERE a.AMT_INCOME_TOTAL IS NOT NULL
GROUP BY faixa_renda_usd
ORDER BY MIN(a.AMT_INCOME_TOTAL/80);
"""

## Proxy de esforço: parcela/renda × taxa
query = """
SELECT
  CASE
    WHEN a.AMT_ANNUITY IS NULL OR a.AMT_INCOME_TOTAL IS NULL OR a.AMT_INCOME_TOTAL=0 THEN 'sem_dado'
    WHEN a.AMT_ANNUITY/NULLIF(a.AMT_INCOME_TOTAL,0) < 0.10 THEN '<10%'
    WHEN a.AMT_ANNUITY/NULLIF(a.AMT_INCOME_TOTAL,0) < 0.20 THEN '10-20%'
    WHEN a.AMT_ANNUITY/NULLIF(a.AMT_INCOME_TOTAL,0) < 0.30 THEN '20-30%'
    ELSE '>=30%'
  END AS faixa_parcela_renda,
  COUNT(*)             AS clientes,
  AVG(a.TARGET*1.0)    AS taxa_inad
FROM application_train a
GROUP BY faixa_parcela_renda
ORDER BY faixa_parcela_renda;
"""

resultado = run_sql(query); resultado

# Inadimplência é mais alta quando parcela consome 20–30% da renda; menor em <10% e ≥30%
